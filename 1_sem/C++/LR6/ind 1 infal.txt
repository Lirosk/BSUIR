	for (int i = 0; i < sizeof(string); i++)
	{
		current = string[i];
		{
			for (int j = i; j < sizeof(string); j++)
			{
				if (current == string[j])
				{
					len += 1;
				}
				else
				{

					if (len > 1 & len < supalen)
					{
						supalen = len;
						sequence = current;
					}
					i += len;
					len = 0;
					break;
				}
			}
		}
	}
